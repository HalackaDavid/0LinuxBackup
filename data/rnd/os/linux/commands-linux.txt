	tar {
		tar -c -z -f <name>tar.gz <path where to be archieved> ---> tar -c -z -f /backup/home_backup.tar.gz /home/user
		-c ---> create a new archive
		-z ---> Compress the archive using gzip
		-f <name>tar.gz ---> Specufues the name and location of the output archive


		tar -x -z -f <name>tar.gz -C <Restor path> ---> tar -x -z -f /backup/home_backup.tar.gz -C /restore_directory
		-x ---> extracts files from the archive
		-z ---> Decompresses the archieve using gzip
		-f ---> Specifies the archive to extract
		-C <restor path> --->  Changes to <restor path> before extracting files. This ensures that the extracted files are placed in <restor path>.
	}
